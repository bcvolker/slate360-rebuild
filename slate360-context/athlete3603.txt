Core Types
// Athlete (player) in a team
type AthletePosition = 'QB' | 'WR' | 'RB' | 'TE' | 'OL' | 'DL' | 'LB' | 'DB' | 'K' | 'P' | 'Other';

interface Athlete {
  id: string;
  orgId: string;
  teamId: string;
  number: number;
  name: string;
  position: AthletePosition;
}

// Practice / session
interface AthleteSession {
  id: string;
  orgId: string;
  teamId: string;
  name: string; // e.g. "Tue Practice â€“ QB Room"
  date: string; // ISO
}

// A tagged rep or play
type ClipStatus = 'processing' | 'ready' | 'error';

interface AthleteClip {
  id: string;
  sessionId: string;
  primaryAthleteId?: string;
  label: string; // e.g. "QB Drop #1"
  status: ClipStatus;
  hasBodyLines: boolean;     // skeleton data available
  hasInteractive3D: boolean; // volumetric data available
  durationSeconds?: number;
}

// Threaded coach/athlete messages on a clip
type ThreadAuthorType = 'coach' | 'athlete';

interface ClipThreadMessage {
  id: string;
  clipId: string;
  authorType: ThreadAuthorType;
  authorName: string;
  createdAt: string;
  text: string;
  mediaUrl?: string; // e.g. selfie or response clip
}

// Homework / participation on a clip
type HomeworkStatus = 'not_assigned' | 'assigned' | 'in_progress' | 'completed';

interface ClipHomeworkStatus {
  clipId: string;
  athleteId: string;
  status: HomeworkStatus;
  watchPercent: number; // 0â€“100
  lastViewedAt?: string;
}

API Endpoints (REST-ish)

All are under /api/athlete360/*.

Sessions

GET /api/athlete360/sessions
  -> AthleteSession[] for current org/team

POST /api/athlete360/sessions
  body: { name: string }
  -> created AthleteSession


Clips

GET /api/athlete360/sessions/:sessionId/clips
  -> AthleteClip[]

POST /api/athlete360/tag
  body: { sessionId: string, primaryAthleteId?: string }
  -> { clipId: string }  // backend handles rolling buffer + processing kickoff


Viewer state

GET /api/athlete360/clips/:clipId
  -> AthleteClip + minimal viewing metadata (urls to Body Lines / Interactive3D assets)


Threads (coach â†” athlete)

GET /api/athlete360/clips/:clipId/threads
  -> ClipThreadMessage[]

POST /api/athlete360/clips/:clipId/threads
  body: { text: string, mediaUrl?: string }
  -> created ClipThreadMessage


Homework & participation

GET /api/athlete360/clips/:clipId/homework
  -> ClipHomeworkStatus[]  // per athlete

POST /api/athlete360/clips/:clipId/homework/assign
  body: { athleteIds: string[] }
  -> { ok: true }

POST /api/athlete360/clips/:clipId/homework/progress
  body: { clipId: string, watchPercent: number }
  // called from the viewer as the player watches
  -> { status: HomeworkStatus } // server decides if 100% => completed

POST /api/athlete360/clips/:clipId/homework/remind
  body: { athleteIds: string[] }
  -> { ok: true } // server sends emails / notifications


This gives you two-way comms and homework tracking in a very clear way.
